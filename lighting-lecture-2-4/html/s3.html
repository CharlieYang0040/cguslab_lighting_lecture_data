maya_basics.py
# Python으로 Maya 자동화 시작!
import maya.cmds as cmds
# 1. 변수 선언
cube_name = "myCube"
cube_size = 2.0
positions = [0, 2, 4, 6, 8]
# 2. 조건문 활용
if cmds.objExists(cube_name):
    cmds.delete(cube_name)
    print("기존 큐브 삭제됨")
# 3. 반복문으로 다중 오브젝트 생성
for i, pos in enumerate(positions):
    # 큐브 생성
    new_cube = cmds.polyCube(
        name=f"{cube_name}_{i}",
        width=cube_size,
        height=cube_size,
        depth=cube_size
    )[0]
    # 위치 설정
    cmds.move(pos, 0, 0, new_cube)
print("자동화 완료! 🎉")
