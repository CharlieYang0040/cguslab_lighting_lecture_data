# Maya Python API 실습 - 오브젝트 생성과 제어
import maya.cmds as cmds
# 1. 기본 오브젝트 생성
cube = cmds.polyCube(name="myCube", width=2)[0]
sphere = cmds.polySphere(name="mySphere", radius=1.5)[0]
# 2. 위치 변경
cmds.move(0, 3, 0, cube) # 큐브를 위로
cmds.move(4, 0, 0, sphere) # 구를 오른쪽으로
# 3. 회전 및 스케일
cmds.rotate(45, 0, 45, cube)
cmds.scale(1.5, 0.5, 1.5, sphere)
# 4. 속성 변경
cmds.setAttr(cube + ".visibility", 1)
cmds.setAttr(sphere + ".displayType", 2) # 템플릿 모드
# 5. 오브젝트 선택
cmds.select([cube, sphere])
print("오브젝트 생성 및 조작 완료!")
