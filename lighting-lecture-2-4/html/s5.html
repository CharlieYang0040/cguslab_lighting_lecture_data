# Python으로 오브젝트 자동 배치 실습
import maya.cmds as cmds
import math
# 1. 선형 배치 - 큐브 10개를 일렬로
def create_linear_cubes():
    spacing = 3.0 # 간격
    for i in range(10):
        # 큐브 생성
        cube = cmds.polyCube(name=f"cube_{i}")[0]
        # X축 방향으로 배치
        cmds.move(i * spacing, 0, 0, cube)
# 2. 그리드 배치 - 5x5 격자 패턴
def create_grid_spheres():
    rows, cols = 5, 5
    spacing = 2.5
    
    for row in range(rows):
        for col in range(cols):
            # 구 생성
            sphere = cmds.polySphere(
                name=f"sphere_{row}_{col}",
                radius=0.8
            )[0]
            
            # 격자 위치 계산
            x = col * spacing
            z = row * spacing
            cmds.move(x, 0, z, sphere)
# 3. 원형 배치 - 원 둘레에 배치
def create_circle_cylinders():
    count = 12 # 12개 실린더
    radius = 8.0 # 원의 반지름
    
    for i in range(count):
        # 각도 계산 (360도를 count로 나눔)
        angle = (i / count) * 360.0
        radian = math.radians(angle)
        
        # 실린더 생성
        cylinder = cmds.polyCylinder(
            name=f"cylinder_{i}",
            height=3.0
        )[0]
        
        # 원형 좌표 계산
        x = math.cos(radian) * radius
        z = math.sin(radian) * radius
        cmds.move(x, 0, z, cylinder)
# 실행!
create_linear_cubes()
create_grid_spheres()
create_circle_cylinders()
print("자동 배치 완료! 총 47개 오브젝트 생성됨 🎯")
